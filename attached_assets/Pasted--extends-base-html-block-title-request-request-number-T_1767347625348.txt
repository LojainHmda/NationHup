/**
 * SERVICE TAB LOGIC (Global vs Specific)
 */
function switchServiceTab(serviceName, rowId = null) {
    const targetPanel = document.getElementById('service-panel-' + serviceName);
    const tabContainer = document.getElementById('service-tab-content');
    const activeBtn = document.querySelector('.service-btn[data-service="' + serviceName + '"]');
    
    // Update Context
    currentServiceContext = {
        serviceName: serviceName,
        rowId: rowId
    };

    if (targetPanel && tabContainer) {
        const isCurrentlyActive = activeBtn && activeBtn.classList.contains('active-tab');
        
        // Reset all tabs to inactive state
        document.querySelectorAll('.service-btn').forEach(btn => {
            btn.classList.remove('active-tab', 'font-semibold');
            btn.style.border = 'none';
            btn.style.color = '#64748b';
            const icon = btn.querySelector('i');
            if (icon) icon.style.color = '#94a3b8';
        });
        
        // Hide all panels inside container
        tabContainer.querySelectorAll('.service-panel').forEach(panel => {
            panel.classList.add('hidden');
        });
        
        if (isCurrentlyActive && rowId === null) {
            tabContainer.classList.add('hidden');
            resetContext();
        } else {
            if (!tabContainer.contains(targetPanel)) {
                tabContainer.appendChild(targetPanel);
            }
            tabContainer.classList.remove('hidden');
            targetPanel.classList.remove('hidden');
            
            activeBtn.classList.add('active-tab', 'font-semibold');
            activeBtn.style.border = '2px solid #3b82f6';
            activeBtn.style.color = '#3b82f6';
            const activeIcon = activeBtn.querySelector('i');
            if (activeIcon) activeIcon.style.color = '#3b82f6';

            updateServiceContextUI(serviceName, rowId);
        }
    }
}

// Updates Badge text and Form Inputs based on Context (Global vs Specific)
function updateServiceContextUI(serviceName, rowId) {
    const badge = document.getElementById('panel-context-badge-' + serviceName);
    if (!badge) return;

    if (rowId) {
        badge.textContent = `Day Specific (ID: ${rowId})`;
        badge.className = "text-xs font-semibold px-2 py-0.5 rounded bg-blue-100 text-blue-700";
        populateFormWithDayData(serviceName, rowId);
    } else {
        badge.textContent = "Global (Applies to All)";
        badge.className = "text-xs font-normal px-2 py-0.5 rounded bg-slate-100 text-slate-500";
        clearServiceForm(serviceName);
    }
    
    if (serviceName === 'hotel') {
        initializeRoomDistribution();
    }
}

function resetContext() {
    currentServiceContext = { serviceName: null, rowId: null };
}

function populateFormWithDayData(service, rowId) {
    const row = document.querySelector(`tr[data-row-id="${rowId}"]`);
    if (!row) return;

    if (service === 'hotel') {
        setInput('hotel_name', row.dataset.hotelName);
        setInput('hotel_category', row.dataset.hotelCat);
    } else if (service === 'transport') {
        setInput('transport_vehicle', row.dataset.transportVehicle);
        setInput('transport_pickup', row.dataset.transportPickup);
        setInput('transport_dropoff', row.dataset.transportDropoff);
    } else if (service === 'guide') {
        setInput('guide_name', row.dataset.guideName);
    } else if (service === 'meal') {
        setInput('meal_restaurant', row.dataset.mealRestaurant);
        setInput('meal_type', row.dataset.mealType);
    }
}

function setInput(id, value) {
    const el = document.getElementById(id);
    if (el) el.value = value || '';
}

function clearServiceForm(service) {
    document.querySelectorAll(`[data-service="${service}"]`).forEach(el => {
        if (el.tagName === 'INPUT' && el.type !== 'date') {
            el.value = '';
        }
        if (el.tagName === 'SELECT') {
            el.selectedIndex = 0;
        }
    });
}

}

function calculateDays() {
    const fromDate = document.getElementById('fromDate').value;
    const toDate = document.getElementById('toDate').value;
    if (fromDate && toDate) {
        const from = new Date(fromDate);
        const to = new Date(toDate);
        const diffTime = Math.abs(to - from);
        const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)) + 1;
        document.getElementById('noOfDays').value = diffDays;
        autoSaveChanges();
    }
}

// Enhanced autoSaveServiceData function
function autoSaveServiceData(serviceType) {
    showAutoSaveIndicator();
    const activePanel = document.getElementById(`service-panel-${serviceType}`);
    const formData = {};
    
    // Collect form data
    const fields = activePanel 
        ? activePanel.querySelectorAll(`[data-service="${serviceType}"]`)
        : document.querySelectorAll(`[data-service="${serviceType}"]`);
    
    fields.forEach(field => {
        if (field.id) {
            formData[field.id] = field.value;
        }
    });
    
    // Add global date range if applicable
    if (currentServiceContext.rowId === null) {
        const globalFrom = document.getElementById(`${serviceType}_global_from`);
        const globalTo = document.getElementById(`${serviceType}_global_to`);
        
        if (globalFrom && globalTo) {
            if (!globalFrom.value || !globalTo.value) {
                showNotification('Please select both From Date and To Date for global services', 'error');
                return;
            }
            formData.global_from = globalFrom.value;
            formData.global_to = globalTo.value;
        }
    }
    
    const payload = {
        service_type: serviceType,
        data: formData
    };

    // Add context information
    if (currentServiceContext.rowId) {
        payload.row_id = currentServiceContext.rowId;
        payload.is_global = false;
    } else {
        payload.is_global = true;
    }
    
    fetch(`/inbound/api/${requestId}/save-service-data`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-CSRFToken': csrfToken
        },
        body: JSON.stringify(payload)
    })
    .then(response => response.json().then(data => ({ ok: response.ok, data })))
    .then(({ ok, data }) => {
        if (ok && data.success) {
            clearInlineError(serviceType);
            showNotification(data.message || 'Service data saved!', 'success');
            
            // Update itinerary if needed
            if (data.itinerary_html) {
                const tbody = document.querySelector('#itineraryTable tbody');
                if (tbody) tbody.innerHTML = data.itinerary_html;
            }
            
            // Update service table - this is the key part
            updateServiceTable(serviceType, data);
            
            // Update summary counts
            updateSummaryCounts();
            
            // Activate icon if needed
            if (currentServiceContext.rowId) {
                activateIcon(currentServiceContext.rowId, serviceType);
            }
        } else {
            const errorMsg = data.error || 'Error saving service data';
            showInlineError(serviceType, errorMsg);
        }
    })
    .catch(error => {
        console.error('Error:', error);
        showNotification('Error saving service data', 'error');
    });
}

// Function to update service tables
function updateServiceTable(serviceType, data) {
    const tableMap = {
        'hotel': { tbodyId: 'hotels-table-body', containerId: 'hotels-table-container' },
        'arrival': { tbodyId: 'arrival-table-body', containerId: 'arrival-table-container' },
        'departure': { tbodyId: 'departure-table-body', containerId: 'departure-table-container' },
        'guide': { tbodyId: 'guides-table-body', containerId: 'guides-table-container' },
        'transport': { tbodyId: 'transports-table-body', containerId: 'transports-table-container' },
        'meal': { tbodyId: 'meals-table-body', containerId: 'meals-table-container' }
    };
    
    const tableInfo = tableMap[serviceType];
    if (!tableInfo) return;
    
    const tbody = document.getElementById(tableInfo.tbodyId);
    const container = document.getElementById(tableInfo.containerId);
    
    // Show the table container when data is added
    if (container) {
        container.style.display = 'block';
    }
    
    // Handle different response formats
    if (data.service_entries_html) {
        if (tbody) {
            tbody.insertAdjacentHTML('beforeend', data.service_entries_html);
        }
    } else if (data.service_entry) {
        if (tbody) {
            const rowHtml = createServiceRowHtml(serviceType, data.service_entry);
            tbody.insertAdjacentHTML('beforeend', rowHtml);
        }
    } else if (data.all_entries_html) {
        if (tbody) {
            tbody.innerHTML = data.all_entries_html;
        }
    }
    
    // Mark summary tables as stale so they refresh when opened
    if (typeof markSummaryTablesStale === 'function') {
        markSummaryTablesStale();
    }
}

// Function to update summary counts
function updateSummaryCounts() {
    const hotelCount = document.querySelectorAll('#hotels-table-body tr').length;
    const hotelSummary = document.getElementById('summary_hotels');
    if (hotelSummary) hotelSummary.textContent = hotelCount;
    
    const transportCount = document.querySelectorAll('#transports-table-body tr').length;
    const transportSummary = document.getElementById('summary_transport');
    if (transportSummary) transportSummary.textContent = transportCount;
    
    const guideCount = document.querySelectorAll('#guides-table-body tr').length;
    const guideSummary = document.getElementById('summary_guides');
    if (guideSummary) guideSummary.textContent = guideCount;
    
    const mealCount = document.querySelectorAll('#meals-table-body tr').length;
    const mealSummary = document.getElementById('summary_meals');
    if (mealSummary) mealSummary.textContent = mealCount;
    
    const flightsContainer = document.getElementById('summary_flights');
    if (flightsContainer) {
        const arrivals = Array.from(document.querySelectorAll('#arrival-table-body tr')).map(tr => {
            const date = tr.cells[0].textContent.trim();
            return `<span class="inline-flex items-center gap-1 bg-green-100 text-green-700 px-1.5 py-0.5 rounded text-[10px]"><i class="fas fa-plane-arrival"></i> ${date}</span>`;
        }).join('');
        
        const departures = Array.from(document.querySelectorAll('#departure-table-body tr')).map(tr => {
            const date = tr.cells[0].textContent.trim();
            return `<span class="inline-flex items-center gap-1 bg-orange-100 text-orange-700 px-1.5 py-0.5 rounded text-[10px]"><i class="fas fa-plane-departure"></i> ${date}</span>`;
        }).join('');
        
        flightsContainer.innerHTML = arrivals + departures || '<span class="text-slate-400 text-[10px]">No flights added</span>';
    }
    
    const hotelsContainer = document.getElementById('summary_hotel_list');
    if (hotelsContainer) {
        const hotels = Array.from(document.querySelectorAll('#hotels-table-body tr')).map(tr => {
            const name = tr.cells[0].textContent.trim();
            const nights = tr.cells[2].textContent.trim();
            return `<span class="inline-flex items-center gap-1 bg-blue-100 text-blue-700 px-1.5 py-0.5 rounded text-[10px]"><i class="fas fa-hotel"></i> ${name} <span class="text-blue-500">(${nights}N)</span></span>`;
        }).join('');
        
        hotelsContainer.innerHTML = hotels || '<span class="text-slate-400 text-[10px]">No hotels added</span>';
    }
    
    // Update tab colors based on data
    updateTabColors();
}

        },

            count: document.querySelectorAll('#guides-table-body tr').length,

            bgHas: 'bg-amber-100', textHas: 'text-amber-700', iconHas: 'text-amber-500', badgeBg: 'bg-amber-500'
        }
    };
    
    for (const [service, config] of Object.entries(tabConfigs)) {
        const tab = document.getElementById(`tab-${service}`);
        if (!tab) continue;
        
        const icon = tab.querySelector('i');
        let badge = tab.querySelector('.tab-count-badge');

        tab.classList.remove('bg-white', 'text-slate-500', config.bgHas, config.textHas);
        if (icon) icon.classList.remove('text-slate-400', config.iconHas);

            tab.classList.add(config.bgHas, config.textHas);
            if (icon) icon.classList.add(config.iconHas);
            
            // Add or update badge
            if (!badge) {
                badge = document.createElement('span');
                badge.className = `ml-1 ${config.badgeBg} text-white text-[10px] px-1.5 rounded-full tab-count-badge`;
                tab.appendChild(badge);
            }
            badge.textContent = config.count;
            badge.className = `ml-1 ${config.badgeBg} text-white text-[10px] px-1.5 rounded-full tab-count-badge`;
        } else {
            // No data - apply default styling
            tab.classList.add('bg-white', 'text-slate-500');
            if (icon) icon.classList.add('text-slate-400');
            
            // Remove badge
            if (badge) badge.remove();
        }
    }
}

// Function to toggle full summary section
let summaryTablesLoaded = false;
function toggleFullSummary() {
    const section = document.getElementById('fullSummarySection');
    const btnText = document.getElementById('toggleSummaryText');
    
    if (section.classList.contains('hidden')) {
        section.classList.remove('hidden');
        btnText.textContent = 'Hide Summary';
        // Only update tables on first open or when data has changed
        if (!summaryTablesLoaded) {
            updateFullSummaryTables();
            summaryTablesLoaded = true;
        }
    } else {
        section.classList.add('hidden');
        btnText.textContent = 'View Full Summary';
    }
}

// Manual save function called by Save buttons
function saveServiceData(serviceType) {
    // Clear inputs after save logic is handled in generic function, 
    // but we explicitly clear here to encourage multiple entries
    if (serviceType === 'hotel') {
        const validation = validateRoomDistribution();
        if (!validation.valid) {
            showInlineError('hotel', validation.message);
            return;
        }
    }
    
    // Perform Save
    autoSaveServiceData(serviceType);
    
    // IMPORTANT: Clear form fields after saving to allow adding another item easily
    setTimeout(() => {
        // Only clear if it was a global save (rowId is null)
        if (!currentServiceContext.rowId) {
            clearServiceForm(serviceType);
            showNotification('Saved. You can add another item.', 'success');
        }
    }, 500);
}

function deleteFlightService(flightType, serviceId) {
    fetch(`/inbound/api/${requestId}/delete-service`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-CSRFToken': csrfToken
        },
        body: JSON.stringify({
            service_type: flightType,
            service_id: serviceId
        })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            showNotification(data.message || 'Flight service deleted!', 'success');
            
            // Find and remove the row by looking for the delete button with matching ID
            const rows = document.querySelectorAll(`#${flightType}-table-body tr`);
            rows.forEach(row => {
                const deleteBtn = row.querySelector(`button[onclick*="${serviceId}"]`);
                if (deleteBtn) {
                    row.remove();
                }
            });
            
            // Check if table is empty and hide container
            const tbody = document.getElementById(`${flightType}-table-body`);
            if (tbody && tbody.children.length === 0) {
                const container = document.getElementById(`${flightType}-table-container`);
                if (container) container.style.display = 'none';
            }
            
            // Update summary counts
            updateSummaryCounts();
            
            // Mark summary tables as stale
            if (typeof markSummaryTablesStale === 'function') {
                markSummaryTablesStale();
            }
        } else {
            showNotification('Error deleting flight service: ' + (data.error || ''), 'error');
        }
    })
    .catch(error => {
        console.error('Error:', error);
        showNotification('Error deleting flight service', 'error');
    });
}

function showNotification(message, type) {
    document.querySelectorAll('.tw-notification').forEach(n => n.remove());
    
    const notification = document.createElement('div');
    let bgColor = 'bg-green-500';
    let icon = 'check';
    
    if (type === 'error') {
        bgColor = 'bg-red-500';
        icon = 'exclamation-circle';
    } else if (type === 'info') {
        bgColor = 'bg-blue-500';
        icon = 'spinner fa-spin';
    }
    
    notification.className = `tw-notification fixed top-20 right-4 px-4 py-2 rounded-lg shadow-lg z-50 ${bgColor} text-white flex items-center gap-2 text-sm`;
    notification.innerHTML = `<i class="fas fa-${icon}"></i><span>${message}</span>`;
    document.body.appendChild(notification);
    
    if (type !== 'info') {
        setTimeout(() => notification.remove(), 3000);
    }
}

function saveAndRefreshItinerary() {
    isSaving = true;
    const data = {
        from_date: document.getElementById('fromDate')?.value || '',
        to_date: document.getElementById('toDate')?.value || '',
        pax: document.querySelector('[name="pax"]')?.value || '',
        customer_type: document.querySelector('[name="customer_type"]')?.value || '',
        contact_name: document.querySelector('[name="contact_name"]')?.value || '',
        nationality: document.querySelector('[name="nationality"]')?.value || ''
    };
    
    showAutoSaveIndicator();
    
    fetch(`/inbound/api/${requestId}/auto-save-and-regenerate`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-CSRFToken': csrfToken
        },
        body: JSON.stringify(data)
    })
    .then(response => response.json())
    .then(data => {
        isSaving = false;
        if (data.success) {
            const daysField = document.getElementById('noOfDays');
            if (daysField) daysField.value = data.no_of_days;
            
            if (data.itinerary_html) {
                const tableBody = document.getElementById('itineraryTableBody');
                const itineraryTable = document.getElementById('itineraryTable');
                const emptyState = document.getElementById('itineraryEmptyState');
                
                if (tableBody) tableBody.innerHTML = data.itinerary_html;
                if (itineraryTable) itineraryTable.classList.remove('hidden');
                if (emptyState) emptyState.classList.add('hidden');
                
                showNotification(data.dates_changed ? 'Itinerary regenerated!' : 'Saved!', 'success');
            } else {
                showNotification('Saved!', 'success');
            }
        } else {
            showNotification(data.message || 'Error saving', 'error');
        }
    })
    .catch(error => {
        isSaving = false;
        console.error('Error:', error);
        showNotification('Error saving', 'error');
    });
}

function exportToExcel() {
    // Gather data from Trip Summary Card (DOM elements)
    // We use textContent to ensure it matches what user sees exactly
    
    // 1. Basic Info
    const requestNumber = "{{ request.request_number }}";
    const customerName = document.querySelector('#selectedCustomerName')?.value || "N/A";
    const dates = `${document.getElementById('fromDate').value} - ${document.getElementById('toDate').value}`;
    const pax = document.querySelector('[name="pax"]').value;
    const days = document.getElementById('noOfDays').value;

    // 2. Counts
    const hotelsCount = document.getElementById('summary_hotels').innerText;
    const transportsCount = document.getElementById('summary_transport').innerText;
    const guidesCount = document.getElementById('summary_guides').innerText;
    const mealsCount = document.getElementById('summary_meals').innerText;

    // 3. Helper to extract text from badges
    const extractItems = (containerId) => {
        const container = document.getElementById(containerId);
        if (!container) return [];
        return Array.from(container.children).map(span => span.innerText.replace(/[\n\r]/g, ' ').trim());
    };

    const flights = extractItems('summary_flights');
    const hotelsList = extractItems('summary_hotel_list');

    // 4. Construct Data for Excel (English Keys Only)
    const excelData = [
        { "Section": "Request Details", "Item": "Request Number", "Value": requestNumber },
        { "Section": "Request Details", "Item": "Customer Name", "Value": customerName },
        { "Section": "Request Details", "Item": "Travel Dates", "Value": dates },
        { "Section": "Request Details", "Item": "Total Days", "Value": days },
        { "Section": "Request Details", "Item": "Total PAX", "Value": pax },
        {}, // Empty Row
        { "Section": "Service Summary", "Item": "Hotels", "Value": hotelsCount },
        { "Section": "Service Summary", "Item": "Transports", "Value": transportsCount },
        { "Section": "Service Summary", "Item": "Guides", "Value": guidesCount },
        { "Section": "Service Summary", "Item": "Meals", "Value": mealsCount },
        {}, // Empty Row
    ];

    // 5. Add Flights
    if (flights.length > 0) {
        excelData.push({ "Section": "Flights", "Item": "Flight Details", "Value": flights.join(', ') });
    } else {
        excelData.push({ "Section": "Flights", "Item": "Flight Details", "Value": "No flights added" });
    }
    excelData.push({});

    // 6. Add Hotels
    if (hotelsList.length > 0) {
        hotelsList.forEach(h => {
            excelData.push({ "Section": "Hotel List", "Item": "Hotel Booking", "Value": h });
        });
    } else {
        excelData.push({ "Section": "Hotel List", "Item": "Hotel Booking", "Value": "No hotels added" });
    }

    // 7. Generate Excel File
    const ws = XLSX.utils.json_to_sheet(excelData);
    
    // Set Column Widths
    const wscols = [
        {wch: 20}, // Section
        {wch: 25}, // Item
        {wch: 50}  // Value
    ];
    ws['!cols'] = wscols;

    const wb = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(wb, ws, "Trip Summary");

    // Filename
    XLSX.writeFile(wb, `Trip_Summary_${requestNumber}.xlsx`);
}

// Global variable to track manual mode state
let isManualMode = false;

/**
 * Helper function to safely get the PAX value from the main request form.
 * This ensures we always get the correct number for calculations.
 */
function getPaxValue() {
    // Select the specific PAX input from the master request card
    const paxField = document.querySelector('form#editMode input[name="pax"]');
    if (!paxField) return 0;

    let val = parseInt(paxField.value);
    return isNaN(val) ? 0 : val;
}

/**
 * Event Listener for Global PAX Input.
 * Updates the room wizard automatically when the total PAX changes.
 */
const globalPaxInput = document.querySelector('form#editMode input[name="pax"]');
if (globalPaxInput) {
    globalPaxInput.addEventListener('input', function() {
        // Update the PAX display inside the hotel wizard
        updateWizardPreviews();
        
        // Recalculate total guest capacity based on rooms
        calculateTotalGuests();

        // If NOT in manual mode, recalculate room distribution based on new PAX
        if (!isManualMode) {
             // Determine the currently selected mode (Economic/DBL/Privacy)
             const currentMode = document.querySelector('.wizard-check:checked')?.name.replace('room_mode_', '') || 'economic';
             applyWizardLayout(currentMode);
        }
    });
}

function calculateEconomicLayout(pax) {
    const trp = Math.floor(pax / 3);
    const afterTrp = pax - (trp * 3);
    const dbl = Math.floor(afterTrp / 2);
    const sgl = afterTrp - (dbl * 2);
    return { sgl, dbl, trp, oth: 0 };
}

function calculateDblLayout(pax) {
    const dbl = Math.floor(pax / 2);
    const sgl = pax % 2;
    return { sgl, dbl, trp: 0, oth: 0 };
}

function calculatePrivacyLayout(pax) {
    return { sgl: pax, dbl: 0, trp: 0, oth: 0 };
}

function formatLayoutPreview(layout) {
    const parts = [];
    if (layout.trp > 0) parts.push(`${layout.trp} TRP`);
    if (layout.dbl > 0) parts.push(`${layout.dbl} DBL`);
    if (layout.sgl > 0) parts.push(`${layout.sgl} SGL`);
    if (layout.oth > 0) parts.push(`${layout.oth} OTH`);
    return parts.length > 0 ? parts.join(' + ') : '--';
}

function updateWizardPreviews() {
    const pax = getPaxValue();
    const paxDisplay = document.getElementById('hotel_pax_display');
    if (paxDisplay) paxDisplay.textContent = pax;
    
    // Since preview elements weren't in HTML, we skip updating them or assume they exist if user adds them
    // The main input fields update via applyWizardLayout
}

function applyWizardLayout(mode) {
    const pax = getPaxValue();
    let layout;
    
    switch(mode) {
        case 'economic': layout = calculateEconomicLayout(pax); break;
        case 'dbl': layout = calculateDblLayout(pax); break;
        case 'privacy': layout = calculatePrivacyLayout(pax); break;
        default: return;
    }
    
    // Add null checks to prevent errors when elements don't exist yet
    const sglField = document.getElementById('hotel_single_rooms');
    const dblField = document.getElementById('hotel_double_rooms');
    const trpField = document.getElementById('hotel_triple_rooms');
    const othField = document.getElementById('hotel_other_rooms');
    
    if (sglField) sglField.value = layout.sgl;
    if (dblField) dblField.value = layout.dbl;
    if (trpField) trpField.value = layout.trp;
    if (othField) othField.value = layout.oth;
    
    if (isManualMode) exitManualMode();
    setActiveWizardMode(mode);
    calculateTotalGuests();
    clearAllErrors();
    
    // Show room guest names section if there are rooms
    if (layout.sgl > 0 || layout.dbl > 0 || layout.trp > 0) {
        showRoomGuestNames();
    }
}

function showRoomGuestNames() {
    const roomGuestNames = document.getElementById('room-guest-names');
    if (roomGuestNames) {
        roomGuestNames.style.display = 'block';
        generateRoomCards();
    }
}

function generateRoomCards() {
    const container = document.getElementById('room-allocation-container');
    if (!container) return;
    
    container.innerHTML = '';
    
    const sglRooms = parseInt(document.getElementById('hotel_single_rooms').value) || 0;
    const dblRooms = parseInt(document.getElementById('hotel_double_rooms').value) || 0;
    const trpRooms = parseInt(document.getElementById('hotel_triple_rooms').value) || 0;
    
    // Generate single room cards
    for (let i = 1; i <= sglRooms; i++) {
        const roomCard = createRoomCard('Single', 1, i);
        container.appendChild(roomCard);
    }
    
    // Generate double room cards
    for (let i = 1; i <= dblRooms; i++) {
        const roomCard = createRoomCard('Double', 2, i);
        container.appendChild(roomCard);
    }
    
    // Generate triple room cards
    for (let i = 1; i <= trpRooms; i++) {
        const roomCard = createRoomCard('Triple', 3, i);
        container.appendChild(roomCard);
    }
}

function createRoomCard(type, capacity, roomNumber) {
    const roomCard = document.createElement('div');
    roomCard.className = 'room-card';
    
    const roomHeader = document.createElement('div');
    roomHeader.className = 'room-card-header';
    
    const roomType = document.createElement('div');
    roomType.className = 'room-type';
    roomType.textContent = `${type} Room ${roomNumber}`;
    
    const roomCapacity = document.createElement('div');
    roomCapacity.className = 'room-capacity';
    roomCapacity.textContent = `Capacity: ${capacity}`;
    
    roomHeader.appendChild(roomType);
    roomHeader.appendChild(roomCapacity);
    
    const roomGuests = document.createElement('div');
    roomGuests.className = 'room-guests';
    
    // Create guest input fields
    for (let i = 1; i <= capacity; i++) {
        const guestInput = document.createElement('input');
        guestInput.type = 'text';
        guestInput.className = 'guest-input';
        guestInput.placeholder = `Guest ${i} Name`;
        guestInput.dataset.roomType = type;
        guestInput.dataset.roomNumber = roomNumber;
        guestInput.dataset.guestNumber = i;
        
        roomGuests.appendChild(guestInput);
    }
    
    roomCard.appendChild(roomHeader);
    roomCard.appendChild(roomGuests);
    
    return roomCard;
}

function calculateTotalGuests() {
    const singleRooms = parseInt(document.getElementById('hotel_single_rooms')?.value) || 0;
    const doubleRooms = parseInt(document.getElementById('hotel_double_rooms')?.value) || 0;
    const tripleRooms = parseInt(document.getElementById('hotel_triple_rooms')?.value) || 0;
    const otherRooms = parseInt(document.getElementById('hotel_other_rooms')?.value) || 0;
    const otherCapacity = parseInt(document.getElementById('hotel_other_capacity')?.value) || 1;
    
    const totalGuests = (singleRooms * 1) + (doubleRooms * 2) + (tripleRooms * 3) + (otherRooms * otherCapacity);
    
    const totalGuestsEl = document.getElementById('hotel_total_guests');
    const capacityNote = document.getElementById('room-capacity-note');
    const pax = getPaxValue();
    
    if (totalGuestsEl) {
        totalGuestsEl.textContent = totalGuests;
        
        if (totalGuests === pax && totalGuests > 0) {
            totalGuestsEl.style.backgroundColor = '#22c55e';
            totalGuestsEl.classList.add('text-white');
            if (capacityNote) capacityNote.textContent = '';
        } else if (totalGuests !== pax && pax > 0) {
            totalGuestsEl.style.backgroundColor = '#ef5350';
            totalGuestsEl.classList.add('text-white');
            if (capacityNote) {
                const diff = totalGuests - pax;
                capacityNote.textContent = diff > 0 ? `${diff} EXTRA` : `${Math.abs(diff)} SHORT`;
                capacityNote.style.color = '#ef5350';
            }
        } else {
            totalGuestsEl.style.backgroundColor = '#e2e8f0';
            totalGuestsEl.classList.remove('text-white');
            totalGuestsEl.classList.add('text-slate-700');
            if (capacityNote) capacityNote.textContent = '';
        }
    }
    return totalGuests;
}

function validateRoomDistribution() {
    const pax = getPaxValue();
    const totalGuests = calculateTotalGuests();
    
    if (totalGuests !== pax) {
        return { valid: false, message: `Room capacity (${totalGuests}) must equal Total Guests (${pax})` };
    }
    return { valid: true };
}

        if (dblField) dblField.value = layout.dbl;
        if (trpField) trpField.value = layout.trp;
    }
    calculateTotalGuests();
}

        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'X-CSRFToken': csrfToken },
        body: JSON.stringify({ name: hotelName, city: hotelCity })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            const select = document.getElementById('hotel_name');
            let optgroup = select.querySelector(`optgroup[label="${hotelCity}"]`);
            if (!optgroup) {
                const separator = select.querySelector('optgroup[label="──────────"]');
                optgroup = document.createElement('optgroup');
                optgroup.label = hotelCity;
                select.insertBefore(optgroup, separator);
            }
            const option = document.createElement('option');
            option.value = hotelName;
            option.textContent = hotelName;
            optgroup.appendChild(option);
            select.value = hotelName;
            closeAddHotelModal();
            showNotification('Hotel added successfully', 'success');
        } else {
            document.getElementById('add-hotel-error').textContent = data.error || 'Error adding hotel';
            document.getElementById('add-hotel-error').classList.remove('hidden');
        }
    })
    .catch(error => {
        document.getElementById('add-hotel-error').textContent = 'Error adding hotel';
        document.getElementById('add-hotel-error').classList.remove('hidden');
    });
}

// Status Updates
const STATUS_ORDER = ['REQUEST', 'SUPPLIER_CONFIRMED', 'QUOTED', 'CONFIRMED', 'INVOICE', 'PROCESSING', 'COMPLETED'];
function updateSidebarState(newStatus) {
    const statusIndex = STATUS_ORDER.indexOf(newStatus);
    document.querySelectorAll('.status-step').forEach(step => {
        const stepStatus = step.dataset.status;
        const stepIndex = STATUS_ORDER.indexOf(stepStatus);
        step.classList.remove('active', 'completed');
        if (stepIndex < statusIndex) step.classList.add('completed');
        else if (stepIndex === statusIndex) step.classList.add('active');
    });
    
    const updateBadge = (id, status) => {
        const badge = document.getElementById(id);
        if(badge) {
            badge.className = 'status-badge status-' + status.toLowerCase().replace('_', '-');
            badge.textContent = status.replace('_', ' ').replace(/\b\w/g, l => l.toUpperCase());
        }
    };
    updateBadge('headerStatusBadge', newStatus);
    // Assuming bottom badge exists
    const bottomBadge = document.querySelector('.stage-bar .status-badge');
    if(bottomBadge) {
        bottomBadge.className = 'status-badge status-' + newStatus.toLowerCase().replace('_', '-');
        bottomBadge.textContent = newStatus.replace('_', ' ').replace(/\b\w/g, l => l.toUpperCase());
    }
}

function updateRequestStatus(status) {
    fetch(`/inbound/api/${requestId}/update-status`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'X-CSRFToken': csrfToken },
        body: JSON.stringify({ status: status })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            updateSidebarState(status);
            showNotification('Status updated', 'success');
            const select = document.getElementById('confirmStageSelect');
            if (select) select.value = '';
        } else {
            showNotification(data.message || 'Error', 'error');
        }
    })
    .catch(e => showNotification('Error', 'error'));
}

function confirmStageChange() {
    const select = document.getElementById('confirmStageSelect');
    if (select.value) updateRequestStatus(select.value);
}

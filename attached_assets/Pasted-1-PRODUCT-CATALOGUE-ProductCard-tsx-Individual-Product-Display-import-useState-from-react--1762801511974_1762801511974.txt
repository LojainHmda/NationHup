ProductCard.tsx (Individual Product Display)
import { useState } from 'react';
import { Plus, Check } from 'lucide-react';
interface ProductColor {
  name: string;
  hex: string;
  imageUrl: string;
}
interface ProductCardProps {
  sku: string;
  name: string;
  price: number;
  originalPrice?: number;
  colors: ProductColor[];
  sizes: string[];
  availableSizes: Record<string, number>;
  isInCart?: boolean;
  onAddToCart?: (color: string) => void;
  cartProductKeys?: Set<string>;
}
export default function ProductCard({
  sku,
  name,
  price,
  originalPrice,
  colors,
  sizes,
  availableSizes,
  isInCart = false,
  onAddToCart,
  cartProductKeys = new Set(),
}: ProductCardProps) {
  const [selectedColor, setSelectedColor] = useState(colors[0]);
  const handleAddToCart = (color: string) => {
    if (onAddToCart) {
      onAddToCart(color);
    }
  };
  return (
    <div className="space-y-3">
      <div className="relative aspect-square bg-card rounded-md group">
        <div className="absolute inset-0 rounded-md overflow-hidden">
          <img
            src={selectedColor.imageUrl}
            alt={name}
            className="w-full h-full object-cover"
          />
        </div>
        
        <button
          onClick={() => handleAddToCart(selectedColor.name)}
          className="absolute top-2 right-2 w-10 h-10 bg-background border-2 border-foreground rounded-full flex items-center justify-center transition-all shadow-lg z-10 hover-elevate active-elevate-2"
          data-testid="button-add-to-cart-main"
        >
          {cartProductKeys.has(`${sku}-${selectedColor.name}`) ? (
            <Check className="w-5 h-5 text-primary" />
          ) : (
            <Plus className="w-5 h-5" />
          )}
        </button>
      </div>
      <div className="space-y-2">
        <div className="flex gap-2">
          {colors.map((color) => {
            const productKey = `${sku}-${color.name}`;
            const isColorInCart = cartProductKeys.has(productKey);
            return (
              <div key={color.name} className="relative">
                <button
                  onClick={() => setSelectedColor(color)}
                  className={`w-12 h-12 rounded-md border-2 transition-all overflow-hidden hover-elevate active-elevate-2 ${
                    selectedColor.name === color.name
                      ? 'border-foreground ring-2 ring-foreground'
                      : 'border-border'
                  }`}
                  aria-label={color.name}
                  data-testid={`color-${color.name.toLowerCase()}`}
                >
                  <img
                    src={color.imageUrl}
                    alt={color.name}
                    className="w-full h-full object-cover"
                  />
                </button>
                
                <button
                  onClick={() => handleAddToCart(color.name)}
                  className="absolute -top-1 -right-1 w-5 h-5 bg-background border-2 border-foreground rounded-full flex items-center justify-center shadow-md z-10 hover-elevate active-elevate-2"
                  data-testid={`button-add-${color.name.toLowerCase()}`}
                >
                  {isColorInCart ? (
                    <Check className="w-3 h-3 text-primary" />
                  ) : (
                    <Plus className="w-3 h-3" />
                  )}
                </button>
              </div>
            );
          })}
        </div>
        <div>
          <p className="text-xs text-muted-foreground uppercase tracking-wide">{sku}</p>
          <h3 className="text-sm font-medium">{name}</h3>
        </div>
        <div className="flex items-center gap-2">
          <span className="font-bold">${price.toFixed(2)}</span>
          {originalPrice && (
            <span className="text-sm text-muted-foreground line-through">
              ${originalPrice.toFixed(2)}
            </span>
          )}
        </div>
        <div>
          <p className="text-xs font-medium uppercase text-muted-foreground mb-1.5">Available Sizes</p>
          <div className="flex flex-wrap gap-1.5">
            {sizes.map((size) => {
              const stock = availableSizes[`${selectedColor.name}-${size}`] || 0;
              const isAvailable = stock > 0;
              return (
                <div
                  key={size}
                  className={`px-2 py-1 text-xs border border-border rounded-md ${
                    !isAvailable ? 'opacity-50 line-through text-muted-foreground' : ''
                  }`}
                  data-testid={`size-${size.toLowerCase()}`}
                >
                  {size}
                </div>
              );
            })}
          </div>
        </div>
      </div>
    </div>
  );
}

import ProductCard from './ProductCard';
interface ProductColor {

}

}
interface ProductGridProps {
  products: ProductData[];

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
      {products.map((product) => (

            onAddToCart?.(product.sku, color);
          }}
        />
      ))}
    </div>
  );
}

2. CART MATRIX TABLE
CartTable.tsx (Size Matrix with Drag-to-Select)
import { useState } from 'react';
import { Checkbox } from '@/components/ui/checkbox';
import { Input } from '@/components/ui/input';
import { Button } from '@/components/ui/button';
import { Trash2 } from 'lucide-react';
interface CartProduct {
  id: string;
  sku: string;
  name: string;
  color: string;
  imageUrl: string;
  price: number;
  sizes: string[];
  quantities: Record<string, number>;
  availableSizes: Record<string, number>;
}
interface CartTableProps {
  products: CartProduct[];
  allSizes: string[];
  onQuantityChange?: (productId: string, size: string, quantity: number) => void;
  onRemoveProduct?: (productId: string) => void;
  onToggleSelect?: (productId: string, selected: boolean) => void;
}
export default function CartTable({
  products,
  allSizes,
  onQuantityChange,
  onRemoveProduct,
  onToggleSelect,
}: CartTableProps) {
  const [selectedProducts, setSelectedProducts] = useState<Set<string>>(new Set(products.map(p => p.id)));
  const [selectedCells, setSelectedCells] = useState<Set<string>>(new Set());
  const [isDragging, setIsDragging] = useState(false);
  const [batchQuantity, setBatchQuantity] = useState('');
  const handleSelectAll = (checked: boolean) => {
    if (checked) {
      setSelectedProducts(new Set(products.map(p => p.id)));
      products.forEach(p => onToggleSelect?.(p.id, true));
    } else {
      setSelectedProducts(new Set());
      products.forEach(p => onToggleSelect?.(p.id, false));
    }
  };
  const handleToggleProduct = (productId: string, checked: boolean) => {
    const newSelected = new Set(selectedProducts);
    if (checked) {
      newSelected.add(productId);
    } else {
      newSelected.delete(productId);
    }
    setSelectedProducts(newSelected);
    onToggleSelect?.(productId, checked);
  };
  const getCellKey = (productId: string, size: string) => `${productId}-${size}`;
  const handleMouseDown = (productId: string, size: string, isAvailable: boolean) => {
    if (!isAvailable) return;
    setIsDragging(true);
    setSelectedCells(new Set([getCellKey(productId, size)]));
  };
  const handleMouseEnter = (productId: string, size: string, isAvailable: boolean) => {
    if (!isDragging || !isAvailable) return;
    setSelectedCells(prev => new Set([...Array.from(prev), getCellKey(productId, size)]));
  };
  const handleMouseUp = () => {
    setIsDragging(false);
  };
  const handleBatchQuantitySubmit = () => {
    const qty = parseInt(batchQuantity) || 0;
    if (qty >= 0 && selectedCells.size > 0) {
      selectedCells.forEach(cellKey => {
        const [productId, size] = cellKey.split('-').slice(0, 2);
        const remainingParts = cellKey.split('-').slice(2);
        const actualSize = remainingParts.length > 0 ? remainingParts.join('-') : size;
        onQuantityChange?.(productId, actualSize, qty);
      });
      setBatchQuantity('');
      setSelectedCells(new Set());
    }
  };
  const handleClearSelection = () => {
    setSelectedCells(new Set());
    setBatchQuantity('');
  };
  const totalQuantity = products.reduce((sum, product) => {
    return sum + Object.values(product.quantities).reduce((s, q) => s + q, 0);
  }, 0);
  const totalPrice = products.reduce((sum, product) => {
    const productTotal = Object.values(product.quantities).reduce((s, q) => s + q, 0);
    return sum + (productTotal * product.price);
  }, 0);
  return (
    <div className="space-y-4" onMouseUp={handleMouseUp} onMouseLeave={handleMouseUp}>
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <Checkbox
            checked={selectedProducts.size === products.length && products.length > 0}
            onCheckedChange={handleSelectAll}
            data-testid="checkbox-select-all"
          />
          <span className="text-sm font-medium uppercase">Select All</span>
        </div>
        {selectedCells.size > 0 && (
          <div className="flex items-center gap-2">
            <span className="text-sm text-muted-foreground">
              {selectedCells.size} cells selected
            </span>
            <Input
              type="number"
              min="0"
              placeholder="Enter qty"
              value={batchQuantity}
              onChange={(e) => setBatchQuantity(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === 'Enter') {
                  handleBatchQuantitySubmit();
                }
              }}
              className="w-24 h-8"
              data-testid="input-batch-quantity"
            />
            <Button
              size="sm"
              onClick={handleBatchQuantitySubmit}
              data-testid="button-apply-batch"
            >
              Apply
            </Button>
            <Button
              size="sm"
              variant="outline"
              onClick={handleClearSelection}
              data-testid="button-clear-selection"
            >
              Clear
            </Button>
          </div>
        )}
      </div>
      <div className="border border-border rounded-md overflow-hidden">
        <div className="overflow-x-auto">
          <table className="w-full">
            <thead className="bg-muted/50 border-b border-border">
              <tr>
                <th className="w-12 p-3"></th>
                <th className="text-left p-3 text-xs font-medium uppercase min-w-[200px]">Product</th>
                {allSizes.map((size) => (
                  <th key={size} className="text-center p-3 text-xs font-medium uppercase w-16">
                    {size}
                  </th>
                ))}
                <th className="text-right p-3 text-xs font-medium uppercase w-20">Total</th>
                <th className="text-right p-3 text-xs font-medium uppercase w-24">Price</th>
                <th className="w-12 p-3"></th>
              </tr>
            </thead>
            <tbody>
              {products.map((product) => {
                const productQuantity = Object.values(product.quantities).reduce((s, q) => s + q, 0);
                const productPrice = productQuantity * product.price;
                return (
                  <tr
                    key={product.id}
                    className="border-b border-border"
                    data-testid={`cart-product-${product.id}`}
                  >
                    <td className="p-3">
                      <Checkbox
                        checked={selectedProducts.has(product.id)}
                        onCheckedChange={(checked) => handleToggleProduct(product.id, checked === true)}
                        data-testid={`checkbox-${product.id}`}
                      />
                    </td>
                    <td className="p-3">
                      <div className="flex items-center gap-3">
                        <div className="w-16 h-16 bg-card rounded-md overflow-hidden flex-shrink-0">
                          <img
                            src={product.imageUrl}
                            alt={product.name}
                            className="w-full h-full object-cover"
                          />
                        </div>
                        <div>
                          <p className="text-xs text-muted-foreground uppercase">{product.sku}</p>
                          <h4 className="text-sm font-medium">{product.name}</h4>
                          <p className="text-xs text-muted-foreground">{product.color}</p>
                        </div>
                      </div>
                    </td>
                    {allSizes.map((size) => {
                      const stock = product.availableSizes[size] || 0;
                      const quantity = product.quantities[size] || 0;
                      const isAvailable = stock > 0;
                      const cellKey = getCellKey(product.id, size);
                      const isSelected = selectedCells.has(cellKey);
                      return (
                        <td
                          key={size}
                          className={`p-1 text-center transition-colors relative ${
                            isSelected ? 'bg-primary/20 border-2 border-primary' : ''
                          }`}
                          data-testid={`cell-${product.id}-${size}`}
                        >
                          {isAvailable ? (
                            <>
                              <div
                                className="absolute inset-0 z-10"
                                onMouseDown={(e) => {
                                  e.preventDefault();
                                  handleMouseDown(product.id, size, isAvailable);
                                }}
                                onMouseEnter={() => {
                                  handleMouseEnter(product.id, size, isAvailable);
                                }}
                                style={{ pointerEvents: isDragging ? 'auto' : 'none' }}
                              />
                              <Input
                                type="number"
                                min="0"
                                value={quantity || ''}
                                onChange={(e) => {
                                  const newQty = parseInt(e.target.value) || 0;
                                  onQuantityChange?.(product.id, size, newQty);
                                }}
                                onFocus={() => setIsDragging(false)}
                                className="w-14 h-8 text-center text-sm relative z-0"
                                data-testid={`input-${product.id}-${size}`}
                              />
                            </>
                          ) : (
                            <span className="text-xs text-destructive">100+</span>
                          )}
                        </td>
                      );
                    })}
                    <td className="p-3 text-right">
                      <span className="text-sm font-medium" data-testid={`total-qty-${product.id}`}>
                        {productQuantity}
                      </span>
                    </td>
                    <td className="p-3 text-right">
                      <span className="text-sm font-bold" data-testid={`total-price-${product.id}`}>
                        ${productPrice.toFixed(2)}
                      </span>
                    </td>
                    <td className="p-3">
                      <Button
                        size="icon"
                        variant="ghost"
                        className="h-8 w-8"
                        onClick={() => onRemoveProduct?.(product.id)}
                        data-testid={`button-remove-${product.id}`}
                      >
                        <Trash2 className="w-4 h-4" />
                      </Button>
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      </div>
      <div className="flex items-center justify-between p-4 bg-muted/30 rounded-md">
        <div className="text-sm">
          <span className="font-medium">Total Quantity of Products: </span>
          <span className="font-bold">{totalQuantity}</span>
        </div>
        <div className="text-sm">
          <span className="font-medium">Total Price: </span>
          <span className="font-bold text-lg">${totalPrice.toFixed(2)}</span>
        </div>
      </div>
    </div>
  );
}

CartSidebar.tsx (Sliding Cart Container)
import { X } from 'lucide-react';
import { Button } from '@/components/ui/button';
import CartTable from './CartTable';
interface CartProduct {
  id: string;
  sku: string;
  name: string;
  color: string;
  imageUrl: string;
  price: number;
  sizes: string[];
  quantities: Record<string, number>;
  availableSizes: Record<string, number>;
}
interface CartSidebarProps {
  isOpen: boolean;
  onClose: () => void;
  products: CartProduct[];
  allSizes: string[];
  onQuantityChange?: (productId: string, size: string, quantity: number) => void;
  onRemoveProduct?: (productId: string) => void;
  onToggleSelect?: (productId: string, selected: boolean) => void;
}
export default function CartSidebar({
  isOpen,
  onClose,
  products,
  allSizes,
  onQuantityChange,
  onRemoveProduct,
  onToggleSelect,
}: CartSidebarProps) {
  return (
    <>
      {/* Backdrop */}
      {isOpen && (
        <div
          className="fixed inset-0 bg-background/80 backdrop-blur-sm z-40 transition-opacity"
          onClick={onClose}
          data-testid="cart-backdrop"
        />
      )}
      {/* Sidebar */}
      <div
        className={`fixed left-0 top-0 h-full w-full max-w-6xl bg-background border-r border-border z-50 transform transition-transform duration-300 ease-in-out ${
          isOpen ? 'translate-x-0' : '-translate-x-full'
        }`}
        data-testid="cart-sidebar"
      >
        <div className="flex flex-col h-full">
          {/* Header */}
          <div className="flex items-center justify-between p-6 border-b border-border">
            <div>
              <h2 className="text-2xl font-bold uppercase">Shopping Cart</h2>
              <p className="text-sm text-muted-foreground">Active cart: adidas w ftw</p>
            </div>
            <Button
              size="icon"
              variant="ghost"
              onClick={onClose}
              data-testid="button-close-cart"
            >
              <X className="w-5 h-5" />
            </Button>
          </div>
          {/* Cart Content */}
          <div className="flex-1 overflow-auto p-6">
            {products.length === 0 ? (
              <div className="flex flex-col items-center justify-center h-full text-center">
                <p className="text-lg font-medium mb-2">Your cart is empty</p>
                <p className="text-muted-foreground mb-6">Add products to get started</p>
                <Button onClick={onClose}>Browse Catalog</Button>
              </div>
            ) : (
              <CartTable
                products={products}
                allSizes={allSizes}
                onQuantityChange={onQuantityChange}
                onRemoveProduct={onRemoveProduct}
                onToggleSelect={onToggleSelect}
              />
            )}
          </div>
        </div>
      </div>
    </>
  );
}
